[Unit]
Description=gunicorn daemon
Requires=gunicorn.socket
After=network.target

[Service]
PIDFile=/run/gunicorn/pid
User=sclay
Group=www-data
WorkingDirectory=/srv/comfortmaps
RuntimeDirectory=gunicorn
ExecStart=/home/sclay/.virtualenvs/comfortmaps/bin/gunicorn --pid /srv/comfortmaps/logs/gunicorn.pid   \
          --bind unix:/srv/comfortmaps/logs/gunicorn.socket comfortmaps.wsgi
ExecReload=/bin/kill -s HUP $MAINPID
ExecStop=/bin/kill -s TERM $MAINPID
PrivateTmp=true

[Install]
WantedBy=multi-user.target
